	tags start with #, such as:
    #NAME - tell server client's name
	?	#HELLO - client to server establish connection
	?	#BYE - client quit chat
		#LIST - list all the people in the chat
		#RMUSER:<name> - remove user from user list on client
    #ADDUSER:<name> - add user to user list on client
    #WISP:<name>:<msg> - whisper to someone

		@USERNAME - talk to some user privately
